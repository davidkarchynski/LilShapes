PROGRAM     := SHAPEDEC*
SHAPEDEC    := SHAPE PROPERTY*
SHAPE       := "ellipse" | "circle" | "rectangle" | "line"
PROPERTY    := POSITION | WIDTH | HEIGHT | RADIUS | TO | FROM | COLOR | LINECOLOR
POSITION    := "pos:" VALUE, VALUE
WIDTH       := "width:" VALUE
HEIGHT      := "height:" VALUE
RADIUS      := "radius:" VALUE
TO          := "to:" VALUE, VALUE
FROM        := "from:" VALUE, VALUE
COLOR       := "color:" COLOR_VAL
COLOR_VAL   := /X11 color name/
VALUE       := \d+(?:\.\d+)?
